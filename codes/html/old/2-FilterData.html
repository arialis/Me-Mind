Filter by coordinates¶

Filtro i dati in formato json ottenuti nella prima fase, in base alle loro coordinate: se si trovano all'interno di un poligono predefinito (l'area che ci interessa), allora li filtro, li pulisco e prendo solo le informazioni a noi utili (nome utente, coordinate, ecc.) e li inserisco in un dataframe, che chiamo "df".

                if 'coordinates'!="null":

for entry in entries:
    try:
        if entry in fileWithCoord:
            file = gz.open('../uidFiles/'+entry)
            for jsonObj in file:
                jsonRead = json.loads(jsonObj)
                usercontent = jsonRead['user']
                creation = jsonRead['created_at']
                userID = usercontent['id']
                screen_name = usercontent['screen_name']
                tweetID = jsonRead['id']
                txt = jsonRead['text']
                if jsonRead['geo'] != 'null':
                    coordinates = jsonRead['geo']
                    pointCoordinates = Point(coordinates['coordinates']) #creo il punto per ogni set di coordinate
                    #print(pointCoordinates)
                    if (pointCoordinates.within(coords2)): #controllo se il punto è nell'area di Pisa
                        #print('yes')
                        lat = pointCoordinates.x
                        lon = pointCoordinates.y
                        new_row = pd.Series(data={'Screen_name':screen_name, 'UserID':userID, 'TweetID':tweetID, 'Coords':[float(lat)]+[float(lon)], 'Lat':lat, 'Lon':lon, 'Created_At':creation, 'Text':txt}, name='x')
                        df = df.append(new_row) #append row to the dataframe
    except:
        continue
