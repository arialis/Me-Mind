def createFilesWithCoord(entries, folder):
    #creo la lista per appendere i file con i tweets che hanno le coordinate (perch molte sono null)
    fileWithCoord = []
    #creo la lista dei file con le coordinate diverse da null
    for entry in entries:
        try:
            file = gz.open('../'+folder+'/'+entry, "rb")
            #contents = file.read()
            #for jsonObj in contents:
            for jsonObj in file:
                jsonRead = json.loads(jsonObj)
                #print(jsonRead)
                #break
                #for k, v in jsonRead:
                if 'coordinates' in jsonRead:
                    if 'coordinates'!="null": 
                        if entry not in fileWithCoord: #controllo che il file non sia gi nella lista
                            fileWithCoord.append(entry)
        except:
            continue
    return fileWithCoord
